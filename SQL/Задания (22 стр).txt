SELECT Адрес
FROM Поставщики NATURAL JOIN
(SELECT Sum(Количество) As [Колво], NП FROM Товары GROUP BY NП)  T
WHERE  T.Колво = (SELECT Max(Колво) FROM T)
----------------------------------------------------------------------------------
2. 
SELECT Поставщик
FROM Товары NATURAL JOIN Поставщики NATURAL JOIN (SELECT Max(Количество) As [Колво], NП FROM Товары GROUP BY NП)) Т
WHERE Т.Колво = (Select Min(Колво) FROM Т)
----------------------------------------------------------------------------------
3.
SELECT Поставщик
FROM Поставщики NATURAL JOIN Товары NATURAL JOIN (SELECT Sum(Количество) As Колво, NП FROM Товары GROUP BY NП) Т
WHERE Т.Колво > (SELECT 1.1*Avg(Колво) FROM T)
EXCEPT
SELECT Поставщик
FROM Поставщики NATURAL JOIN Товары
WHERE Адрес = "Москва" OR Товар = "Хлеб"
----------------------------------------------------------------------------------
4.
SELECT Отдел
FROM МоиСотрудники
GROUP BY Отдел
HAVING Count(ФИО) > 5
----------------------------------------------------------------------------------
5.
SELECT ФИО
FROM МоиСотрудники
WHERE Начислено = (SELECT Max(Начислено) FROM МоиСотрудники WHERE Отдел <> 1)
WHERE Должность NOT IN ('Директор','Нач%тдела')
----------------------------------------------------------------------------------
6.
SELECT ФИО
FROM МоиСотрудники
WHERE Начислено > (SELECT 1.2*Avg(Начислено) FROM МоиСотрудники WHERE Должность NOT IN ('Директор','Фин%иректор'))
----------------------------------------------------------------------------------
7.
SELECT ФИО
FROM МоиСотрудники NATURAL JOIN (SELECT Отдел, Count(Табномер) as Колво
FROM МоиСотрудники
GROUP BY Отдел) Т
WHERE Т.Колво = (SELECT Max(Колво) FROM T)
AND Должность LIKE "Нач%тдела"
----------------------------------------------------------------------------------
8.
SELECT ФИО
FROM МоиСотрудники NATURAL JOIN (SELECT Отдел, Avg(Начислено) as Нач
FROM МоиСотрудники
GROUP BY Отдел) Т
WHERE Т.Нач = (SELECT Min(Нач) FROM T)
AND Должность LIKE "Нач%тдела"
----------------------------------------------------------------------------------
9.
SELECT ФИО
FROM МоиСотрудники NATURAL JOIN (SELECT Отдел, Min(Начислено) as Мин
FROM МоиСотрудники
WHERE Должность = "Инженер"
GROUP BY Отдел) Т
WHERE Т.Мин = (SELECT Max(Мин) FROM T)
----------------------------------------------------------------------------------
10.
SELECT ФИО, Телефон
FROM МоиСотрудники NATURAL JOIN Отделы
WHERE Должность LIKE "Нач%тдела"
AND Начислено = (SELECT Max(Начислено)
FROM МоиСотрудники
WHERE Должность LIKE "Нач%тдела")
----------------------------------------------------------------------------------
11.
SELECT Заведующий, Характеристика
FROM МоиСотрудники NATURAL JOIN Отделы
WHERE Начислено = (SELECT Min(Начислено)
FROM МоиСотрудники)
----------------------------------------------------------------------------------
12.
SELECT Телефон
FROM МоиСотрудники NATURAL JOIN Отделы
GROUP BY Заведующий
HAVING Count(ФИО) < 6
----------------------------------------------------------------------------------
13.
SELECT Отдел
FROM МоиСотрудники
WHERE Начислено <= (SELECT 1.3 * Avg(Начислено)
FROM МоиСотрудники)
WHERE Должность LIKE "Нач%тдела"
UNION
SELECT Отдел
FROM МоиСотрудники
WHERE Должность LIKE "Инженер"
----------------------------------------------------------------------------------
14.
SELECT ФИО
FROM МоиСотрудники
WHERE Начислено > (SELECT 1.1 * Avg(Начислено)
FROM МоиСотрудники)
INTERSECT
SELECT ФИО
FROM МоиСотрудники
WHERE Отдел <> 2
AND Должность NOT LIKE "Нач%тдела"
